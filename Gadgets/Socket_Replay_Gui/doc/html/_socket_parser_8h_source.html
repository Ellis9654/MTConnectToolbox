67  void split (std::string str, raw_message_t & msg)
68  {
69  const std::string::size_type size(str.size( ));
70  size_t start = 0;
71  size_t range = 0;
72 
73  msg.clear( );
74 
75  /* Explanation:
76  * Range - Length of the word to be extracted without spaces.
77  * start - Start of next word. During initialization, starts at space 0.
78  *
79  * Runs until it encounters a ' ', then splits the string with a substr() function,
80  * as well as making sure that all characters are lower-case (without wasting time
81  * to check if they already are, as I feel a char-by-char check for upper-case takes
82  * just as much time as lowering them all anyway.
83  */
84  size_t i;
85 
86  for ( i = 0; i < str.size( ); i++ )
87  {
88  if ( i == 443 )
89  {
90  i = i;
91  }
92 
93  if ( str[i] == ',' )
94  {
95  std::string buf = str.substr(start, range);
96  int uc;
97  sscanf(buf.c_str( ), "%2X", &uc);
98  msg.push_back((uint8_t) uc);
99  start = i + 1;
100  range = 0;
101  }
102  else
103  {
104  ++range;
105  }
106  }
107 
108  // We always have a trailing , with no data afterward so this is unnecessary
109  // msg.push_back( toLower(str.substr(start, range)) );
110  // skip trailing \n
111  }

119  double ProcessStream ( )
120  {
121  // Get line from "in" stream into string _buffer until line feed found.
122  // If the line feed delimiter is found, it is extracted and discarded
123  // buffer is a std::string so there will be no "overrun"
124  getline(in, _buffer);
125 
126  if ( in.eof( ) )
127  {
128  if ( !Repeat( ) )
129  {
130  return -1;
131  }
132  Reset( );
133  in.clear( ); // forget we hit the end of file
134  in.seekg(0, std::ios::beg); // move to the start of the file
135  getline(in, _buffer);
136  }
137  _linenum++;
138 
139  // Check for empty buffer - return 0 wait
140  if ( _buffer.empty( ) )
141  {
142  return 0;
143  }
144 
145  // Now get timestamp - in field 1
146  size_t delim = _buffer.find("|");
147 
148  if ( delim == std::string::npos )
149  {
150  return 0;
151  }
152 
153  // strip out timestamp from buffer
154  std::string timestamp = _buffer.substr(0, delim);
155 
156  if ( !timestamp.empty( ) )
157  {
158  _lasttime = _currenttime;
159  _currenttime = Timing::GetDateTime(timestamp);
160 
161  if ( _bFirstTime )
162  {
163  _duration = boost::posix_time::seconds(0);
164  _bFirstTime = false;
165  }
166  else
167  {
168  _duration = _currenttime - _lasttime;
169  _realtimetime += _duration;
170  }
171 
172  // in shdr this returns a shdr string with new timestamp.
173 
174  _timestamp = Timing::GetDateTimeString(_realtimetime);
175 
176  // _buffer=_buffer.substr(delim+1); // after the |
177 
178  // For binary hex dump need to translate ascii hex into raw binary.
179  // also need to add timestamp for timing and remove trailing \n
180  split(_buffer.substr(delim + 1), _msg);
181  }
182  else
183  {
184  _duration = boost::posix_time::seconds(0);
185  }
186 
187  // http://www.boost.org/doc/libs/1_31_0/libs/date_time/doc/class_time_duration.html
188  return _duration.total_milliseconds( );
189  }
